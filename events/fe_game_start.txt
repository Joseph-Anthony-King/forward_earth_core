                                }
                            }

                                save_event_target_as = UNoE_npc
                                add_resource = {
                                    minerals = 200
                                    energy = 200
                                    food = 200

                                    consumer_goods = 100

                                }
                            }

                            owner = event_target:UNoE_npc

                            ethos = {
                                ethic = "ethic_xenophile"

                            }
                        }

                        solar_system = {
                            create_starbase = {
                                size = starbase_starport
                                module = shipyard
